# Nginx container deployment
#
# https://levelup.gitconnected.com/mount-files-into-your-kubernetes-deployment-via-configmaps-bedfe5350e25
#
#------------------------------------------------------------------------------

apiVersion: v1
kind: ConfigMap
metadata:
  name: cfgmap-splash-page
  namespace: openedx
data:
  configmap_splash_page: |
    <!DOCTYPE html>
    <html lang="en">
      <head>
        <title>Stepwise Math</title>
      </head>
      <body>
        <h1>Open edX</h1>
        <h2>Looks like you caught us at an awkward moment</h2>
      </body>
    </html>
---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: nginx
  namespace: openedx
  labels:
    app: nginx
  annotations:
    # mcdaniel
    # https://www.cyberciti.biz/faq/nginx-upstream-sent-too-big-header-while-reading-response-header-from-upstream/
    # to fix "[error] 199#199: *15739 upstream sent too big header while reading response header from upstream"
    # ---------------------
    nginx.ingress.kubernetes.io/proxy-busy-buffers-size: "512k"
    nginx.ingress.kubernetes.io/proxy-buffers: "4 512k"
    nginx.ingress.kubernetes.io/proxy-buffer-size: "256k"
spec:
  selector:
    matchLabels:
      app: nginx
  replicas: 2
  template:
    metadata:
      labels:
        app: nginx
    spec:
      containers:
      - name: nginx
        image: nginx:latest
        ports:
        - containerPort: 80
        volumeMounts:
          - name: splash-page
            mountPath: /usr/share/nginx/html/index.html
            subPath: configmap_splash_page
      volumes:
      - name: splash-page
        configMap:
          name: cfgmap-splash-page
